53  static inline uint32_t Murmur2(const void* pdata, uint32_t size,
54  const uint32_t seed)
55  {
56  const unsigned int m = 0x5bd1e995;
57  const int r = 24;
58 
59  // Initialize the hash to a 'random' value
60  uint32_t h = seed ^ size;
61 
62  // Mix 4 bytes at a time into the hash
63  const unsigned char* data = (const unsigned char*)pdata;
64 
65  while(size >= 4)
66  {
67  uint32_t k = *(uint32_t*)data;
68 
69  k *= m;
70  k ^= k >> r;
71  k *= m;
72 
73  h *= m;
74  h ^= k;
75 
76  data += 4;
77  size -= 4;
78  }
79 
80  // Handle the last few bytes of the input array
81  switch(size)
82  {
83  case 3: h ^= data[2] << 16;
84  case 2: h ^= data[1] << 8;
85  case 1: h ^= data[0]; h *= m;
86  };
87 
88  // Do a few final mixes of the hash to ensure the last few
89  // bytes are well-incorporated.
90  h ^= h >> 13;
91  h *= m;
92  h ^= h >> 15;
93 
94  return h;
95  }
