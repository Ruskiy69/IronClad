232 bool CQuadTree::RInsert(obj::CRigidBody* pBody, QTNode* pStart)
233 {
234  if(pBody == NULL || pStart == NULL) return false;
235 
236  // Is this a leaf node?
237  if(pStart->IsLeaf())
238  {
239  // Completely full quad
240  if(!pStart->AddObject(pBody))
241  {
242  // Can we add another branch?
243 
244  // If not, force object into full quad regardless.
245  if(!pStart->CanSplit(QT_MAX_DEPTH))
246  {
247  pStart->AddObject(pBody, true);
248  pBody->pNode = pStart;
249  return true;
250  }
251 
252  // Otherwise, create another branch.
253  else
254  {
255  pStart->Split();
256  return RInsert(pBody, pStart);
257  }
258  }
259 
260  pBody->pNode = pStart;
261  return true;
262  }
263 
264  // Just a branch, has babies.
265  else
266  {
267  for(uint8_t i = 0; i < QTNode::s_SPLIT_SIZE; ++i)
268  {
269  if(pStart->operator[](i)->NodeCollision(pBody))
270  return RInsert(pBody, pStart->operator[](i));
271  }
272  }
273 
274  // This shit shouldn't happen, unless called by Update().
275  // If it does, though, we go up the tree to the root (makes sense lol),
276  // to find a node that we can fit in.
277 
278  // Return false if we've reached the root node and
279  // still can't insert.
280  return (pStart->mp_Parent == NULL) ? false :
281  RInsert(pBody, pStart->mp_Parent);
282 }
