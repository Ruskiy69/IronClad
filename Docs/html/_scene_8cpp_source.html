80 void CScene::Render()
81 {
82  // Model-view matrix.
83  static math::matrix4x4_t MVMatrix = math::IDENTITY;
84 
85  m_ShadowVBO.Clear();
86  m_GeometryVBO.FinalizeBuffer();
87 
88  // Clear swap frame-buffer.
89  m_FBOSwap.Enable();
90  m_FBOSwap.Clear();
91  m_FBOSwap.Disable();
92 
93  // Enable render-to-texture.
94  m_FBO.Enable();
95  m_FBO.Clear();
96 
97  // Clear frame-buffer settings.
98  glClearColor(0.1f, 0.1f, 0.1f, 1.f);
99  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
100 
101  // Blending is essential.
102  glEnable(GL_BLEND);
103  glDisable(GL_DEPTH_TEST);
104 
105  // Normal transparency blending function.
106  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
107 
108  // Bind geometry VBO.
109  m_GeometryVBO.Bind();
110 
111  // Render all of the meshes.
112  for(size_t i = 0; i < mp_sceneObjects.size(); ++i)
113  {
114  // Load the model-view matrix.
115  mp_sceneObjects[i]->GetMesh().LoadPositionMatrix(MVMatrix);
116 
117  // Adjust for the camera.
118  MVMatrix[0][3] += m_Camera.x;
119  MVMatrix[1][3] += m_Camera.y;
120 
121  // Get mesh surfaces.
122  const std::vector<gfx::surface_t*>& meshSurfaces =
123  mp_sceneObjects[i]->GetMesh().GetSurfaces();
124 
125  // Quads get one single texture, this accounts for animation.
126  if(meshSurfaces.size() == 1)
127  {
128  this->StandardRender(mp_sceneObjects[i], MVMatrix);
129  }
130  else
131  {
132  if(!mp_sceneObjects[i]->IsRenderable()) continue;
133 
134  // Render each surface.
135  for(size_t j = 0; j < meshSurfaces.size(); ++j)
136  {
137  this->StandardRender(meshSurfaces[j], MVMatrix);
138  }
139  }
140  }
141 
142  uint32_t final_texture = m_FBO.GetTexture();
143 
144  if(m_lighting)
145  {
146  Globals::g_FullscreenVBO.Bind();
147 
148  // Render effects on the scene.
149  m_FBOSwap.Enable();
150  glBindTexture(GL_TEXTURE_2D, m_FBO.GetTexture());
151 
152  // Additive blending for lighting.
153  glBlendFunc(GL_ONE, GL_ONE);
154 
155  // Render all lights onto the frame-buffer.
156  for(size_t i = 0; i < mp_sceneLights.size(); ++i)
157  {
158  this->LightRender(mp_sceneLights[i]);
159  }
160 
161  // Final FBO texture to render onto the screen.
162  // This texture has all of the post-processing on it.
163  final_texture = m_FBOSwap.GetTexture();
164  }
165 
166  if(m_postfx)
167  {
168  // Render each effect onto the scene.
169  // The FBOs are swapped every time,
170  // adding effects on top of one another.
171  for(size_t i = 0; i < mp_sceneEffects.size(); ++i)
172  {
173  final_texture = this->PostProcessingRender(
174  mp_sceneEffects[i],
175  i % 2 == 0 ? m_FBOSwap : m_FBO,
176  i % 2 == 0 ? m_FBO.GetTexture() : m_FBOSwap.GetTexture());
177  }
178  }
179 
180  // This is so the FBO blends with the data in the default frame-buffer.
181  glDisable(GL_BLEND);
182  glBlendFuncSeparate(GL_ONE, GL_ONE, GL_ZERO, GL_ONE);
183 
184  // Render to the screen now.
185  m_FBOSwap.Disable();
186  Globals::g_FullscreenVBO.Bind();
187 
188  // Draw off-screen texture to screen.
189  Globals::g_DefaultEffect.Enable();
190  Globals::g_DefaultEffect.SetMatrix("mv", math::IDENTITY.GetMatrixPointer());
191 
192  glBindTexture(GL_TEXTURE_2D, final_texture);
193  Globals::g_FullscreenVBO.Draw();
194  glBindTexture(GL_TEXTURE_2D, 0);
195 
196  Globals::g_DefaultEffect.Disable();
197  Globals::g_FullscreenVBO.Unbind();
198 }

214 void CScene::StandardRender(CEntity* pEntity,
215  const math::matrix4x4_t& ModelView)
216 {
217  if(!pEntity->IsRenderable()) return;
218 
219  gfx::surface_t* pSurface = pEntity->GetMesh().GetSurfaces()[0];
220  gfx::material_t* pMaterial = pSurface->pMaterial;
221 
222  // Nothing to render if there's no texture/shader.
223  if(pMaterial == NULL) return;
224 
225  // Bind texture and shader, returns false if no shader,
226  // so bind the default.
227  if(!pMaterial->pShader)
228  {
229  Globals::g_DefaultEffect.Enable();
230  Globals::g_DefaultEffect.SetMatrix("mv", ModelView);
231  }
232  else
233  {
234  pMaterial->pShader->Bind();
235  int mvloc = pMaterial->pShader->GetUniformLocation("mv");
236  int pjloc = pMaterial->pShader->GetUniformLocation("proj");
237 
238  if(mvloc == -1 || pjloc == -1) return;
239 
240  glUniformMatrix4fv(mvloc, 1, GL_TRUE,
241  ModelView.GetMatrixPointer());
242 
243  glUniformMatrix4fv(pjloc, 1, GL_TRUE,
244  m_Window.GetProjectionMatrix());
245  }
246 
247  if(pEntity->GetTexture() != NULL)
248  pEntity->GetTexture()->Bind();
249 
250  // Do rendering.
251  glDrawElements(m_geo_type, pSurface->icount, GL_UNSIGNED_SHORT,
252  (void*)(sizeof(uint16_t)*(pSurface->start)));
253 
254  // Unbind shader / texture.
255  glBindTexture(GL_TEXTURE_2D, 0);
256  Globals::g_DefaultEffect.Disable();
257 }

356 void CScene::UpdateShadows(const math::vector2_t& LightPosition)
357 {
358  // Retrieve buffers from the GPU.
359  vertex2_t* vertices = (vertex2_t*)m_GeometryVBO.GetTemporaryBuffer(
360  GL_ARRAY_BUFFER);
361 
362  uint16_t* indices = (uint16_t*)m_GeometryVBO.GetTemporaryBuffer(
363  GL_ELEMENT_ARRAY_BUFFER);
364 
365  if(!vertices || !indices)
366  {
367  g_Log.Flush();
368  g_Log << "[ERROR] Failed to map vertex buffer.\n";
369  g_Log << "[ERROR] Most likely caused by an improper bind.\n";
370  g_Log << "[ERROR] OpenGL error: " << glGetString(glGetError());
371  g_Log << "\n";
372  g_Log.PrintLastLog();
373 
374  handle_error(g_Log.GetLastLog());
375  }
376 
377  // Determine buffer sizes.
378  uint32_t vsize = m_GeometryVBO.GetVCount();
379  uint32_t isize = m_GeometryVBO.GetICount();
380 
381  if(!vsize || !isize)
382  {
383  g_Log.Flush();
384  g_Log << "[ERROR] Empty buffers.\n";
385  g_Log << "[ERROR] OpenGL error code: " << glGetError() << "\n";
386  g_Log.PrintLastLog();
387  return;
388  }
389 
390  for(size_t j = 0; j < mp_sceneObjects.size(); ++j)
391  {
392  if(!mp_sceneObjects[j]->CastsShadow()) continue;
393 
394  // Find the lowest and highest indices to get the clock-wise ordered
395  // vertices from the buffer.
396  std::vector<gfx::surface_t*>& Surfaces =
397  mp_sceneObjects[j]->GetMesh().GetSurfaces();
398 
399  uint16_t mindex = indices[Surfaces.front()->start];
400  uint16_t maxdex = indices[Surfaces.front()->start];
401 
402  for(size_t k = Surfaces.front()->start;
403  k < Surfaces.back()->start + Surfaces.back()->icount;
404  ++k)
405  {
406  if(indices[k] < mindex) mindex = indices[k];
407  if(indices[k] > maxdex) maxdex = indices[k];
408  }
409 
410  // We need to track which vertex is closest to the light source
411  // in order to make the shadows render in a clock-wise order.
412  uint16_t closest_vertex = 0;
413 
414  math::vector2_t Position = vertices[mindex].Position + m_Camera +
415  mp_sceneObjects[j]->GetMesh().GetPosition();
416 
417  float closest_distance = math::distance(
418  Position.x, Position.y,
419  LightPosition.x, LightPosition.y, false);
420 
421  // Ray-trace valid shadow outlines.
422  for(size_t i = mindex; i <= maxdex; ++i)
423  {
424  math::vector2_t Position = vertices[i].Position + m_Camera +
425  mp_sceneObjects[j]->GetMesh().GetPosition();
426 
427  // Line from the vertex to the next one.
428  math::vector2_t CasterLine =
429  vertices[(i == maxdex) ? 0 : i + 1].Position -
430  vertices[i].Position;
431 
432  // Vector from the light to the vertex.
433  math::vector2_t LightToVertex = Position - LightPosition;
434 
435  // Vector normal to the caster line, facing the right.
436  math::vector2_t Normal(CasterLine.y, -CasterLine.x);
437 
438  // A line will cast a shadow if N . L >= 0
439  //if(Normal * LightToVertex >= 0.f)
440  {
441  vertex2_t Final1, Final2;
442 
443  Final1.Position = Position;
444  Final2.Position = (Position - LightPosition) * 4096.f;
445 
446  m_shadowVertices.reserve(m_shadowVertices.size() + 2);
447  m_shadowVertices.push_back(Final1);
448  m_shadowVertices.push_back(Final2);
449 
450  // Check distance from light.
451  float this_distance =
452  LightToVertex.x * LightToVertex.x +
453  LightToVertex.y * LightToVertex.y;
454 
455  if(closest_vertex == -1)
456  {
457  closest_vertex = m_shadowVertices.size() - 2;
458  closest_distance = this_distance;
459  }
460  else if(this_distance < closest_distance)
461  {
462  closest_distance = this_distance;
463  closest_vertex = m_shadowVertices.size() - 2;
464  }
465  }
466  }
467 
468  // Add indices for the shadow vertices to actually make the shadow.
469  m_shadowIndices.reserve(m_shadowIndices.size() +
470  (3 * m_shadowVertices.size() + 6) / 2);
471 
472  // Add indices for the shadows in a clock-wise fashion.
473  uint8_t counter = 0;
474  size_t j = closest_vertex;
475 
476  for( ; j < m_shadowVertices.size() - 2; ++j)
477  {
478  if(j % 2 == 0)
479  {
480  m_shadowIndices.push_back(j);
481  m_shadowIndices.push_back(j + 1);
482  m_shadowIndices.push_back(j + 2);
483  }
484  else
485  {
486  m_shadowIndices.push_back(j + 1);
487  m_shadowIndices.push_back(j + 2);
488  m_shadowIndices.push_back(j + 3);
489  }
490  }
491 
492  m_shadowIndices.push_back(j);
493  m_shadowIndices.push_back(j + 1);
494  m_shadowIndices.push_back(0);
495  m_shadowIndices.push_back(j + 1);
496  m_shadowIndices.push_back(0);
497  m_shadowIndices.push_back(1);
498 
499  for(size_t i = 0; i + 2 < closest_vertex; ++i)
500  {
501  if(i % 2 == 0)
502  {
503  m_shadowIndices.push_back(i);
504  m_shadowIndices.push_back(i + 1);
505  m_shadowIndices.push_back(i + 2);
506  }
507  else
508  {
509  m_shadowIndices.push_back(i + 1);
510  m_shadowIndices.push_back(i);
511  m_shadowIndices.push_back(i + 2);
512  }
513  }
514  }
515 
516  // Add shadow data to the VBO.
517  m_ShadowVBO.AddData(&m_shadowVertices[0], m_shadowVertices.size(),
518  &m_shadowIndices[0], m_shadowIndices.size());
519 
520  m_shadowIndices.clear();
521  m_shadowVertices.clear();
522 
523  // Done with buffers.
525  glUnmapBuffer(GL_ARRAY_BUFFER);
526  glUnmapBuffer(GL_ELEMENT_ARRAY_BUFFER);
527 }
