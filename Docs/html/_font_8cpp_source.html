16 {
17  g_Log.Flush();
18  g_Log << "[INFO] Loading font: " << filename << "\n";
19  g_Log.PrintLastLog();
20 
21  if(!m_FontRender.Init(gfx::IC_GRAYSCALE)) return false;
22 
23  m_FontRender.Enable();
24  if(!m_FontRender.SetMatrix("mv", math::IDENTITY) ||
25  !m_FontRender.SetMatrix("proj", gfx::CWindow::GetProjectionMatrix()))
26  return false;
27  m_FontRender.Disable();
28 
29  if(FT_New_Face(m_Library, filename.c_str(), 0, &m_FontFace) != 0)
30  return false;
31 
32  // Set the size of the font face.
33  // Since the function expects a size in 1/64 pixels, we multiply
34  // by 64 (same as left-shifting 6 bits) before passing.
35  // The 96 represents a 96-dpi font bitmap.
36  if(FT_Set_Char_Size(m_FontFace, size << 6, size << 6, 96, 96) != 0)
37  return false;
38 
39  // For logging :)
40  std::stringstream ss;
41 
42  // Load all printable characters.
43  // If you visit an ASCII table (like www.asciitable.com) you will see
44  // that the only valid values for printing are the space character all
45  // the way up to the tilde (~).
46  for(size_t i = ' '; i <= '~'; ++i)
47  {
48  FT_Glyph glyph;
49 
50  // Locate the index of the character in the font face.
51  uint32_t index = FT_Get_Char_Index(m_FontFace, i);
52  if(index == 0) continue;
53 
54  // Load the glyph into the font face.
55  FT_Load_Glyph(m_FontFace, index, FT_LOAD_RENDER);
56 
57  // Render the glyph as a mono-chrome bitmap.
58  FT_Render_Glyph(m_FontFace->glyph, FT_RENDER_MODE_NORMAL);
59 
60  // Put the glyph in the glyph slot into an actual glyph struct.
61  FT_Get_Glyph(m_FontFace->glyph, &glyph);
62 
63  FT_GlyphSlot slot = m_FontFace->glyph;
64  FT_Bitmap& bitmap = slot->bitmap;
65 
66  // Bitmap dimensions
67  uint32_t w = bitmap.width;
68  uint32_t h = bitmap.rows;
69 
70  // Create the OpenGL bitmap texture handle.
71  asset::CTexture* pTexture =
72  asset::CAssetManager::Create<asset::CTexture>();
73 
74  // Log filename as font_name:char
75  ss << filename << ":" << (char)i;
76  pTexture->SetFilename(ss.str());
77  ss.str(std::string());
78 
79  // We need to copy the data over to a new buffer in order
80  // to properly pass it to the GPU.
81  unsigned char* data = new unsigned char[w * h];
82  memset(data, NULL, w * h * sizeof(unsigned char));
83  memcpy(data, bitmap.buffer, sizeof(unsigned char) * w * h);
84 
85  // Alignment for pixels needs to be at 1 byte.
86  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
87  pTexture->LoadFromRaw(GL_R8, GL_RED, w, h, data);
88 
89  // Restore default alignment value. I haven't actually tested this
90  // part so it may or may not actually be the default.
91  glPixelStorei(GL_UNPACK_ALIGNMENT, 4);
92 
93  // Delete bitmap buffers
94  delete[] data;
95  FT_Done_Glyph(glyph);
96 
97  Glyph render_glyph;
98  render_glyph.pTexture = pTexture;
99  render_glyph.dim = math::rect_t(w, h,
100  slot->advance.x / 64,
101  slot->metrics.horiBearingY / 64);
102 
103  mp_glyphTextures[i] = render_glyph;
104  }
105 
106  return FT_Done_Face(m_FontFace) == 0;
107 }

109 math::rect_t CFont::RenderText(const std::string& text, const math::vector2_t& Pos)
110 {
111  // VBO that will contain our text vertex data.
112  gfx::CVertexBuffer VBO;
113 
114  // Track total text size.
115  math::rect_t Size(Pos.x, Pos.y, 0, 0);
116 
117  if(!m_loaded || text.empty()) return Size;
118 
119  // Vertex buffer size, index buffer size.
120  uint16_t vlen = text.length() << 2;
121  uint16_t ilen = text.length() * 6;
122 
123  // Create buffers and zero them.
124  vertex2_t* verts = new vertex2_t[vlen];
125  uint16_t* inds = new uint16_t [ilen];
126 
127  memset(inds, NULL, sizeof(uint16_t) * ilen);
128  memset(verts, NULL, sizeof(vertex2_t) * vlen);
129 
130  // Track width and max height.
131  int max_w = 0, max_h = 0;
132 
133  // The x-position to start the next character at.
134  int32_t last_w = Pos.x;
135 
136  // Fill up buffers. Each character needs 4 vertices, so
137  // we increment by 4 each iteration then compensate for
138  // that throughout the loop.
139  for(size_t i = 0; i < vlen; i += 4)
140  {
141  // Retrieve dimensions from the dictionary.
142  // Since we're doing i += 4, the index in the text string
143  // would be text[i / 4].
144  rect_t Dim = mp_glyphTextures[text[i >> 2]].dim;
145 
146  float w = last_w;
147  float h = Dim.y;
148  last_w = w + Dim.w; // Increase until next char by the
149  // bitmap's horizontal advance value.
150 
151  // [i] : top left
152  // [i + 1] : top right
153  // [i + 2] : bottom right
154  // [i + 3] : bottom left
155  verts[i].Position = math::vector2_t(w, Pos.y - Dim.h);
156  verts[i+1].Position = math::vector2_t(last_w, Pos.y - Dim.h);
157  verts[i+2].Position = math::vector2_t(last_w, Pos.y - Dim.h + h);
158  verts[i+3].Position = math::vector2_t(w, Pos.y - Dim.h + h);
159 
160  // Load up the bitmap texture coordinates moving counter-clockwise
161  // from the origin.
162  verts[i].TexCoord = math::vector2_t(0, 0);
163  verts[i+1].TexCoord = math::vector2_t(1, 0);
164  verts[i+2].TexCoord = math::vector2_t(1, 1);
165  verts[i+3].TexCoord = math::vector2_t(0, 1);
166 
167  // The vertices all use the font color.
168  // See CFont::SetColor()
169  for(size_t j = i; j < i + 4; ++j)
170  verts[j].Color = m_Color;
171 
172  // The index in the buffer that we need is i / 4 * 6, since
173  // i / 4 gives us the current character, and each character
174  // needs 6 indices.
175  int x = (i >> 2) * 6;
176 
177  // Assign the values.
178  inds[x] = i;
179  inds[x+1] = i + 1;
180  inds[x+2] = i + 3;
181  inds[x+3] = i + 3;
182  inds[x+4] = i + 2;
183  inds[x+5] = i + 1;
184 
185  // Keep track of the overall dimensions.
186  max_w += Dim.w;
187  max_h = ic::math::max<int>(max_h, Dim.h + h);
188  }
189 
190  // Tally up dimensions.
191  Size.w = max_w;
192  Size.h = max_h;
193 
194  // Enable font-rendering shader.
195  m_FontRender.Enable();
196 
197  // Give data to GPU.
198  VBO.Init();
199  VBO.AddData(verts, vlen, inds, ilen);
200  VBO.FinalizeBuffer();
201  VBO.Bind();
202 
203  glEnable(GL_BLEND);
204  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
205 
206  // Draw each character with its texture enabled.
207  for(size_t i = 0; i < text.length(); ++i)
208  {
209  mp_glyphTextures[text[i]].pTexture->Bind();
210  glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_SHORT,
211  (void*)(sizeof(uint16_t) * i * 6));
212  }
213 
214  glDisable(GL_BLEND);
215 
216  // Delete GPU buffers.
217  VBO.Clear();
218 
219  // Unbind all the things.
220  glBindTexture(GL_TEXTURE_2D, 0);
221  VBO.Unbind();
222  m_FontRender.Disable();
223 
224  // Delete old buffers in RAM.
225  delete[] verts;
226  delete[] inds;
227 
228  return Size;
229 }
